Quick Start

open ReasonNode

Node.run(
  Fs.mkdir(
    "myDir", Fs.(
      fun
      | Ok => prerr_endline("Dir Created!"))
      | Err(e) => prerr_endline("Something went wrong")
    )
  )
);
